193      @Test 
194      public void Response_Ver() { 
195   
196          String currency = "ARS"; 
197   
198          //Request 
199          Response response = 
200          given(). 
201                  auth(). 
202                  oauth2("59f40dae-90ca-c336-95f0-492527e1ff77"). 
203                  log().all(). 
204                  when(). 
205                  get(EndPoint.GET_PROPERTIES); 
206   
207          //Inline Validation ****************************************************************************************** 
208          //Hard Assertion: 1 Fail all Fail 
209          response.then().body("defaultCurrency", equalTo(currency)).body("orderTitleMax", equalTo(99)).log().all(); 
210          //Soft Assertion: 1 Fail rest are verify 
211          //response.then().body("defaultCurrency", equalTo(currency),"orderTitle", equalTo("NONE")); 
212   
213          //Path Validation  ****************************************************************************************** 
214          //Hard Validation 
215          Assert.assertEquals(response.path("defaultCurrency"), currency); 
216          //Soft Assertion 
217          SoftAssert softAssert = new SoftAssert(); 
218          softAssert.assertEquals(response.path("defaultCurrency"), currency); 
219          softAssert.assertEquals(response.path("orderTitle"), "NONE"); 
220          softAssert.assertAll(); 
221      }
